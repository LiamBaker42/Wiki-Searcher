// --- API Helper Functions --- // UPGRADED: Now fetches up to 500 links, unlocking massive "Hub Nodes" async function fetchLinks(pageTitle) { const url = `https://en.wikipedia.org/w/api.php?action=query&titles=${encodeURIComponent(pageTitle)}&prop=links&pllimit=500&format=json&origin=*`; try { const response = await fetch(url); const data = await response.json(); const pages = data.query.pages; const pageId = Object.keys(pages)[0]; if (!pages[pageId].links) return []; // Filter out non-article namespaces (Talk:, Category:, File:, etc.) return pages[pageId].links.filter(link => link.ns === 0).map(link => link.title); } catch (e) { return []; } } // Fetches the summary and image for the final result cards async function fetchPageDetails(pageTitle) { const url = `https://en.wikipedia.org/w/api.php?action=query&titles=${encodeURIComponent(pageTitle)}&prop=pageimages|extracts&exintro&explaintext&pithumbsize=200&format=json&origin=*`; try { const response = await fetch(url); const data = await response.json(); const pages = data.query.pages; const pageId = Object.keys(pages)[0]; const page = pages[pageId]; return { title: page.title, summary: page.extract ? page.extract.substring(0, 120) + '...' : 'No summary available.', image: page.thumbnail ? page.thumbnail.source : 'https://upload.wikimedia.org/wikipedia/commons/8/80/Wikipedia-logo-v2.svg' }; } catch (e) { return { title: pageTitle, summary: 'Error fetching details.', image: '' }; } } // --- Optimized BFS Algorithm --- async function runSolver() { const startInput = document.getElementById('start').value.trim(); const targetInput = document.getElementById('target').value.trim(); const btn = document.getElementById('solveBtn'); const ticker = document.getElementById('ticker'); const results = document.getElementById('results'); if (!startInput || !targetInput) return alert("Please enter both pages."); // UI Reset btn.disabled = true; btn.innerText = "Searching..."; results.innerHTML = ''; ticker.style.display = 'block'; ticker.innerText = `Initializing search from "${startInput}"...`; let queue = [startInput]; let visited = new Set([startInput.toLowerCase()]); let parentMap = {}; let pagesExplored = 0; const MAX_PAGES = 300; // Lower safety limit because each page now brings in 500 links! let found = false; let actualTargetName = targetInput; // Direct match check right at the start if (startInput.toLowerCase() === targetInput.toLowerCase()) { ticker.innerText = `You started at the destination!`; await renderPath({}, startInput, startInput); btn.disabled = false; btn.innerText = "Solve"; return; } while (queue.length > 0 && pagesExplored < MAX_PAGES) { let currentPage = queue.shift(); pagesExplored++; ticker.innerText = `[Explored: ${pagesExplored} pages | Queue Size: ${queue.length}] Scanning: ${currentPage}...`; // Fetch up to 500 neighbors let neighbors = await fetchLinks(currentPage); for (let neighbor of neighbors) { let neighborLower = neighbor.toLowerCase(); if (!visited.has(neighborLower)) { visited.add(neighborLower); parentMap[neighborLower] = currentPage; // Map lowercase child to exact parent string // OPTIMIZATION: Look-ahead verification. Check matches *before* enqueueing. if (neighborLower === targetInput.toLowerCase()) { found = true; actualTargetName = neighbor; // Capture Wikipedia's exact casing ticker.innerText = `Target found! Mapping path...`; queue = []; break; } queue.push(neighbor); } } if (found) break; } if (!found) { ticker.innerText = `Search stopped. Could not find a path within ${pagesExplored} page scans. Try closer concepts!`; } else { ticker.innerText = `Success! Path found after scanning ${pagesExplored} pages.`; await renderPath(parentMap, startInput, actualTargetName); } btn.disabled = false; btn.innerText = "Solve"; } // --- UI Rendering --- async function renderPath(parentMap, start, target) { const resultsContainer = document.getElementById('results'); let path = []; let current = target; // Backtrack using our lowercase-optimized parent map while (current && current.toLowerCase() !== start.toLowerCase()) { path.push(current); current = parentMap[current.toLowerCase()]; } path.push(start); path.reverse(); // Render Clean UI Timeline Cards for (let i = 0; i < path.length; i++) { let details = await fetchPageDetails(path[i]); let cardHTML = `

${details.title}

${details.summary}

`; resultsContainer.innerHTML += cardHTML; if (i < path.length - 1) { resultsContainer.innerHTML += `

`; } } }
