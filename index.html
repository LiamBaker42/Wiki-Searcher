// --- API Helper Functions --- // 1. Forward Search: Gets links ON the current page async function fetchLinks(pageTitle) { const url = `https://en.wikipedia.org/w/api.php?action=query&titles=${encodeURIComponent(pageTitle)}&prop=links&pllimit=500&format=json&origin=*`; try { const response = await fetch(url); const data = await response.json(); const pages = data.query.pages; const pageId = Object.keys(pages)[0]; if (!pages[pageId].links) return []; return pages[pageId].links.filter(link => link.ns === 0).map(link => link.title); } catch (e) { return []; } } // 2. Backward Search: Gets pages that link TO the current page async function fetchBacklinks(pageTitle) { const url = `https://en.wikipedia.org/w/api.php?action=query&titles=${encodeURIComponent(pageTitle)}&prop=linkshere&lhlimit=500&format=json&origin=*`; try { const response = await fetch(url); const data = await response.json(); const pages = data.query.pages; const pageId = Object.keys(pages)[0]; if (!pages[pageId].linkshere) return []; return pages[pageId].linkshere.filter(link => link.ns === 0).map(link => link.title); } catch (e) { return []; } } // 3. UI Data Fetcher async function fetchPageDetails(pageTitle) { const url = `https://en.wikipedia.org/w/api.php?action=query&titles=${encodeURIComponent(pageTitle)}&prop=pageimages|extracts&exintro&explaintext&pithumbsize=200&format=json&origin=*`; try { const response = await fetch(url); const data = await response.json(); const pages = data.query.pages; const pageId = Object.keys(pages)[0]; const page = pages[pageId]; return { title: page.title, summary: page.extract ? page.extract.substring(0, 120) + '...' : 'No summary available.', image: page.thumbnail ? page.thumbnail.source : 'https://upload.wikimedia.org/wikipedia/commons/8/80/Wikipedia-logo-v2.svg' }; } catch (e) { return { title: pageTitle, summary: 'Error fetching details.', image: '' }; } } // --- Core Bidirectional BFS Algorithm --- async function runSolver() { const startInput = document.getElementById('start').value.trim(); const targetInput = document.getElementById('target').value.trim(); const btn = document.getElementById('solveBtn'); const ticker = document.getElementById('ticker'); const results = document.getElementById('results'); if (!startInput || !targetInput) return alert("Please enter both pages."); // Reset UI btn.disabled = true; btn.innerText = "Searching Both Ends..."; results.innerHTML = ''; ticker.style.display = 'block'; if (startInput.toLowerCase() === targetInput.toLowerCase()) { ticker.innerText = `You started at the destination!`; await buildAndRenderBidirectionalPath({}, {}, startInput); btn.disabled = false; btn.innerText = "Solve"; return; } // Set up our two parallel search queues let fwdQueue = [startInput]; let bwdQueue = [targetInput]; // Parent maps: Key is the page (lowercase), Value is where it came from let fwdParents = { [startInput.toLowerCase()]: null }; let bwdParents = { [targetInput.toLowerCase()]: null }; let pagesExplored = 0; const MAX_PAGES = 400; // 200 forward, 200 backward let intersectionNode = null; // Take turns expanding one node forward, then one node backward while (fwdQueue.length > 0 && bwdQueue.length > 0 && pagesExplored < MAX_PAGES) { // --- STEP 1: Expand Forward --- let fwdCurrent = fwdQueue.shift(); pagesExplored++; ticker.innerText = `[Page ${pagesExplored}] Forward scanning: ${fwdCurrent}...`; let fwdNeighbors = await fetchLinks(fwdCurrent); for (let neighbor of fwdNeighbors) { let lower = neighbor.toLowerCase(); if (!(lower in fwdParents)) { fwdParents[lower] = fwdCurrent; // Did we collide with the backward search? if (lower in bwdParents) { intersectionNode = neighbor; break; } fwdQueue.push(neighbor); } } if (intersectionNode) break; // Collision found! Stop searching. // --- STEP 2: Expand Backward --- let bwdCurrent = bwdQueue.shift(); pagesExplored++; ticker.innerText = `[Page ${pagesExplored}] Backward scanning: ${bwdCurrent}...`; let bwdNeighbors = await fetchBacklinks(bwdCurrent); for (let neighbor of bwdNeighbors) { let lower = neighbor.toLowerCase(); if (!(lower in bwdParents)) { bwdParents[lower] = bwdCurrent; // Did we collide with the forward search? if (lower in fwdParents) { intersectionNode = neighbor; break; } bwdQueue.push(neighbor); } } if (intersectionNode) break; // Collision found! Stop searching. } if (!intersectionNode) { ticker.innerText = `Search stopped. Could not bridge the gap within ${MAX_PAGES} total scans.`; } else { ticker.innerText = `Collision detected at "${intersectionNode}"! Reconstructing path...`; await buildAndRenderBidirectionalPath(fwdParents, bwdParents, intersectionNode); ticker.innerText = `Success! Path connected after scanning ${pagesExplored} pages.`; } btn.disabled = false; btn.innerText = "Solve"; } // --- Path Reconstruction & Rendering --- async function buildAndRenderBidirectionalPath(fwdParents, bwdParents, intersectionNode) { const resultsContainer = document.getElementById('results'); // 1. Build path from Start to Intersection let fwdPath = []; let curr = intersectionNode; while (curr) { fwdPath.push(curr); curr = fwdParents[curr.toLowerCase()]; } fwdPath.reverse(); // Flip it so it goes Start -> Intersection // 2. Build path from Intersection to Target let bwdPath = []; // Skip the intersection node itself since it's already at the end of the fwdPath curr = bwdParents[intersectionNode.toLowerCase()]; while (curr) { bwdPath.push(curr); curr = bwdParents[curr.toLowerCase()]; } // Combine them const finalPath = [...fwdPath, ...bwdPath]; // Render Cards for (let i = 0; i < finalPath.length; i++) { let details = await fetchPageDetails(finalPath[i]); let cardHTML = `

${details.title}

${details.summary}

`; resultsContainer.innerHTML += cardHTML; if (i < finalPath.length - 1) { resultsContainer.innerHTML += `
↓
`; } } }
